  99. /**
  100.  * error handling callback function 
  101.  * php4 cannot handle all errors, that's why we have to use a
  102.  * workaround using output buffering (see post by
  103.  * smp at ncoastsoft dot com at
  104.  *   http://www.php.net/manual/en/function.set-error-handler.php
  105.  */ 
  106. function jsonrpc_catch_errors($buffer)
  107. {
  108.     if (ereg("(error</b>:)(.+)(<br)", $buffer, $regs) ) 
  109.     {
  110.         // parse error string from PHP error message 
  111.         $err = preg_replace("/<.*?>/","",$regs[2]);
  112.  
  113.         // return error formatted as a JSONRPC error response
  114.         return
  115.             '{' .
  116.             '  error:' .
  117.             '  {' .
  118.             '    "origin":' . JsonRpcError_Origin_Server . ',' . 
  119.             '    "code":' .  JsonRpcError_ScriptError . ',' .
  120.             '    "message":"Fatal PHP Error: '. addslashes($err) .
  121.             ' "}' .
  122.             '}';
  123.     } 
  124.     else
  125.     {
  126.         // Buffer does not contain a php error message, so return it
  127.         // unmodified.
  128.         return $buffer;
  129.     }
