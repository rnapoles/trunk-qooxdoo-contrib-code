  1485.   }
  1486.   
  1487.   /**
  1488.    * gets a single option value

  1497.   }
  1498.  
  1499.   /**
  1500.    * sets a single option value

  1503.    * @param $value mixed
  1504.    * @param $id int/string[optional]

  1512.   
  1513.   /**
  1514.    * Gets the name of the main data table
  1515.    * @return string
  1516.    */
  1517.   function getTableName()
  1518.   {
  1519.     return $this->table;
  1520.   }
  1521.   
  1522.   /**
  1523.    * returns the prefix for tables used by this

  1528.   function getTablePrefix()
  1529.   {

  1531.     $dsModel =& $this->getDatasourceModel();
  1532.     if ( $dsModel )
  1533.     {
  1534.       $prefix = $dsModel->getTablePrefix();
  1535.     }
  1536.     //$this->info("Prefix for {$this->name} is '$prefix'.");

  1538.   }
  1539.   
  1540.   /**

  1542.    * all needed tables and columns. schema document will be available

  1551.     
  1552.     /*
  1553.      * return if no database connection is present
  1554.      */
  1555.     if ( ! $this->db )
  1556.     {
  1557.       $this->raiseError("Cannot setup schema - no database connection.");
  1558.     }
  1559.     
  1560.     /*
  1561.      * sql type
  1562.      */
  1563.     $sqltype    =  $this->db->getType();
  1564.     
  1565.     /*
  1566.      * the schema xml object and document
  1567.      */

  1569.     $doc        =& $modelXml->getDocument();
  1570.     
  1571.     //$this->info($doc->asXml());
  1572.  
  1573.     /*
  1574.      * model-level attributes
  1575.      */
  1576.     $modelAttrs  = $doc->model->attributes();
  1577.     
  1578.     /*
  1579.      * name, should be a dot-separated, java-like class name
  1580.      */
  1581.     $this->name  = $modelAttrs['name'];
  1582.     
  1583.     /*
  1584.      * class can be specified separately, defaults
  1585.      * to name and converted into PHP-style class name
  1586.      * foo.bar.Baz => foo_bar_baz

  1588.     $this->class = strtolower(str_replace(".","_", either( $modelAttrs['class'], $modelAttrs['name'] ) ) ); 
  1589.     
  1590.     /*
  1591.      * the type can be provided if class is the implementation

  1593.      */
  1594.     $this->type  = $modelAttrs['type'];
  1595.  
  1596.     /*
  1597.      * whether the setup process should upgrade the schema or just use existing tables in
  1598.      * whatever schema they are in
  1599.      */
  1600.     if ( $modelAttrs['upgradeSchema'] == "no" and ! $forceUpgrade )
  1601.     {
  1602.       $this->log("Schema document for model name '{$this->name}' is not to be upgraded.","propertyModel");
  1603.       return null;
  1604.     }    
  1605.  
  1606.     /*
  1607.      * This model doesn't have a table backend
  1608.      */
  1609.     if ( $modelAttrs['table']  == "no" )
  1610.     {
  1611.       $this->log("Model name '{$this->name}' has no table backend.","propertyModel");
  1612.       return null;  
  1613.     }
  1614.     
  1615.     /*
  1616.      * The table name is provided as the 'table' property
  1617.      * of the class, usually prefixed by the datasource name, 
  1618.      * if applicable to the model 
  1619.      */
  1620.     if ( ! $modelAttrs['table'] )
  1621.     {
  1622.       $this->raiseError("Model '{$this->name}': No table name!"); 
  1623.     }
  1624.     $prefix = $this->getTablePrefix();
  1625.     $this->table = $prefix . $modelAttrs['table'];
  1626.  
  1627.     /*
  1628.      * Wheter the table for this model exists already
  1629.      */
  1630.     $tableExists  = $this->db->tableExists($this->table);
  1631.     
  1632.     /*
  1633.      * Get the modelTableInfo persistent object to look up if this
  1634.      * table has been initialized already. To avoid an indefinitive loop,
  1635.      * the qcl_db_PersistentModel used by qcl_db_ModelTableInfo must
  1636.      * be especially treated. You can upgrade its schema only be deleting
  1637.      * the table.
  1638.      */
  1639.     $modelTableInfo = null;
  1640.     if ( $this->instanceOf("qcl_db_PersistentModel") and $tableExists )
  1641.     {
  1642.       $isInitialized = true;
  1643.       $forceUpgrade  = false;
  1644.     }
  1645.     else
  1646.     {
  1647.       /*
  1648.        * cache object globally and in each object to avoid
  1649.        * multiple instantiation of the same object
  1650.        */
  1651.       if ( ! $this->modelTableInfo )
  1652.       {
  1653.         require_once "qcl/db/ModelTableInfo.php";
  1654.         global $qclDbModelTableInfo;
  1655.         if ( ! $qclDbModelTableInfo )
  1656.         {
  1657.           $qclDbModelTableInfo = new qcl_db_ModelTableInfo(&$this); // pass by reference doesn't work, so two objects will always exist 
  1658.         }
  1659.         $this->modelTableInfo =& $qclDbModelTableInfo; 
  1660.       }
  1661.       $datasourceModel =& $this->getDatasourceModel();
  1662.       $isInitialized   = $this->modelTableInfo->isInitialized( &$datasourceModel, $this->table, $this->class, $this->schemaTimestamp );
  1663.     }
  1664.        
  1665.     /*
  1666.      * Return if table exists and schema hasn't changed for the table
  1667.      */
  1668.     
  1669.     if ( $tableExists and $isInitialized and ! $forceUpgrade ) 
  1670.     {
  1671.       $this->log(
  1672.         "Schema document for model name '{$this->name}', ".
  1673.         "type '{$this->type}', class '{$this->class}' hasn't changed.",
  1674.         "propertyModel"
  1675.       );
  1676.       return null;
  1677.     }  
  1678.     
  1679.     /*
  1680.      * update schema
  1681.      */ 
  1682.     $this->info("Creating or updating tables for model name '{$this->name}', type '{$this->type}', class '{$this->class}' ...");
  1683.     
  1684.     /*
  1685.      * create main data table if necessary
  1686.      */
  1687.     $table = $this->table;
  1688.     if ( !$this->db->tableExists($table) )
  1689.     {
  1690.       $this->db->createTable($table);
  1691.       $this->schemaXml->hasChanged = true;
  1692.     }
  1693.    
  1694.     $this->log("Creating aliases...","propertyModel");
  1695.     
  1696.     /*
  1697.      * create alias table
  1698.      */
  1699.     $aliases  = $doc->model->definition->aliases;
  1700.     if ($aliases)
  1701.     {
  1702.       $aliasMap = array();
  1703.       foreach($aliases->children() as $alias)
  1704.       {
  1705.         $a = $alias->attributes();
  1706.         $aliasMap[$a['for']] = $modelXml->getData($alias);
  1707.       }
  1708.     }
  1709.  
  1710.     
  1711.     $this->log("Setting up table columns...","propertyModel");
  1712.     
  1713.     /*
  1714.      * properties as columns
  1715.      */
  1716.     $properties = $doc->model->definition->properties->children(); 
  1717.     foreach($properties as $property)
  1718.     {
  1719.       $attr      = $property->attributes();
  1720.       $propName  = $attr['name'];
  1721.       $colName   = either($aliasMap[$propName],$propName);
  1722.       
  1723.       /*
  1724.        * save property/alias
  1725.        */
  1726.       $this->properties[$propName] = $colName;
  1727.       
  1728.       /*
  1729.        * skip if no column definition available
  1730.        */
  1731.       if ( count($property->children()) == 0)
  1732.       {
  1733.         continue;
  1734.       }
  1735.       
  1736.       /*
  1737.        * @todo: check for <sql type="$sqltype">
  1738.        */
  1739.       if ( ! is_object( $property->sql ) )
  1740.       {
  1741.         $this->warn("Model Property '$propName' has no definition matching the sql type.");
  1742.         continue;
  1743.       }
  1744.  
  1745.       /*
  1746.        * descriptive definition of the existing table schema,
  1747.        */
  1748.       $descriptiveDef = $this->db->getColumnDefinition($table,$colName);
  1749.       
  1750.       /*
  1751.        * normative definition of how the table schema should look like
  1752.        */
  1753.       $normativeDef = trim($modelXml->getData($property->sql)); 
  1754.       
  1755.       //$this->debug("Property '$propName', Column '$colName':");
  1756.       //$this->debug("Old def: '$descriptiveDef', new def:'$normativeDef'");
  1757.  
  1758.       /*
  1759.        * Skip column if descriptive and normative column definition are identical.
  1760.        * removing "on update" part since this won't be present in the 
  1761.        * descriptive definition and "default null" since this is the default anyways.
  1762.        * @todo check for position and auto-increment
  1763.        */
  1764.       $sql1 = trim(preg_replace("/on update .+$|default null/", "", strtolower( $normativeDef ) ) );

  1767.       //$this->log("'$sql1' == '$sql2'? ");
  1768.       
  1769.       /*
  1770.        * continue with the next property if nothing has changed
  1771.        */
  1772.       if ( $sql1 == $sql2 ) continue; 
  1773.       
  1774.       /*
  1775.        * Dropping indexes before changing the schema speeds up things immensely. 
  1776.        * They will be recreated further dow. This must only be done at
  1777.        * the first change of schema and not at all if nothing changes.
  1778.        */
  1779.       if ( ! $indexes )
  1780.       {
  1781.         $indexes =& $doc->model->definition->indexes;
  1782.         
  1783.         if ( $indexes )
  1784.         {
  1785.           foreach ( $indexes->children() as $index )
  1786.           {
  1787.             $attrs   = $index->attributes();
  1788.             $name    = $attrs['name'];
  1789.             $this->db->dropIndex($table,$name);
  1790.           }
  1791.         }
  1792.       }      
  1793.       
  1794.       /*
  1795.        * position
  1796.        */
  1797.       $position  = $attr['position'];
  1798.       if ( $position )
  1799.       {
  1800.         $normativeDef .= " " . $position;
  1801.       }      
  1802.       
  1803.       /*
  1804.        * If column does not exist, add it
  1805.        */      
  1806.       if ( ! $descriptiveDef )
  1807.       {
  1808.         $this->db->addColumn( $table, $colName, $normativeDef );
  1809.       }
  1810.       
  1811.       /*
  1812.        * otherwise, if it exists but is different: modifiy it
  1813.        */      
  1814.       else
  1815.       {
  1816.         $this->db->modifyColumn($table,$colName,$normativeDef);
  1817.       }
  1818.  
  1819.       /* 
  1820.        * unique column
  1821.        */
  1822.       if ( $attr['unique'] == "yes" )
  1823.       {
  1824.         $indexName = $colName . "_unique";
  1825.         if ( ! count( $this->db->getIndexColumns($table,$indexName) ) ) 
  1826.         {
  1827.           $this->db->addIndex("unique", $table, $indexName, $colName ); 
  1828.         }
  1829.       }
  1830.  
  1831.       /*
  1832.        * index 
  1833.        */
  1834.       if ( $attr['index'] == "yes" )
  1835.       {
  1836.         $indexName = $colName;
  1837.         $this->info("Creating index for $colName"); 
  1838.         if ( ! count( $this->db->getIndexColumns($table,$indexName) ) ) 
  1839.         {
  1840.           $this->db->addIndex("index", $table, $indexName, $colName ); 
  1841.         }
  1842.       }       
  1843.       
  1844.     }
  1845.     
  1846.     /*
  1847.      * contraints 
  1848.      */
  1849.     $constraints =& $this->schemaXml->getNode("/model/definition/constraints"); 
  1850.     if ( $constraints )
  1851.     {
  1852.       
  1853.       $this->log("Checking constraints...","propertyModel");
  1854.       
  1855.       foreach ( $constraints->children() as $constraint )
  1856.       {
  1857.         $attrs = $constraint->attributes();
  1858.  
  1859.         switch ($attrs['type'])
  1860.         {
  1861.           /*
  1862.            * primary key(s)
  1863.            */
  1864.           case "primary":
  1865.             $primaryKeys = array();

  1867.             {
  1868.               $a = $property->attributes();
  1869.               $propName = $a['name'];
  1870.               $primaryKeys[] =  either($aliasMap[$propName],$propName);
  1871.             }            
  1872.             
  1873.             // analyze existing primary key
  1874.             $oldPrimaryKeys = $this->db->getPrimaryKey($table);
  1875.             
  1876.             //$this->debug("Old primary key: " . implode(",",$oldPrimaryKeys) );

  1878.  
  1879.             if ( $oldPrimaryKeys != $primaryKeys )
  1880.             {
  1881.               /*
  1882.                * drop old if different from model
  1883.                */
  1884.               if ( $oldPrimaryKeys)
  1885.               {
  1886.                 $this->db->dropPrimaryKey($table);
  1887.               }
  1888.               
  1889.               /*
  1890.                * add new primary key
  1891.                */
  1892.               $this->db->addPrimaryKey($table,$primaryKeys);
  1893.             }
  1894.             break;
  1895.         }
  1896.       }
  1897.     }
  1898.       
  1899.     /*
  1900.      * indexes
  1901.      */
  1902.     $indexes =& $doc->model->definition->indexes;
  1903.     if ( $indexes )
  1904.     {
  1905.       $this->log("Creating indexes ...","propertyModel");
  1906.       
  1907.       foreach ( $indexes->children() as $index )
  1908.       {
  1909.         $attrs        = $index->attributes();
  1910.         $indexType    = strtoupper($attrs['type']);
  1911.         switch ($indexType)
  1912.         {
  1913.           /*
  1914.            * fulltext and unique index
  1915.            */
  1916.           case "FULLTEXT":
  1917.           case "UNIQUE":
  1918.             
  1919.             $indexProperties = array();
  1920.             foreach($index->children() as $property)
  1921.             {
  1922.               $a        = $property->attributes();
  1923.               $propName = $a['name'];
  1924.               $indexProperties[] = either($aliasMap[$propName],$propName) ;
  1925.             }
  1926.             
  1927.             // analyze existing index
  1928.             $indexName    = either($attrs['name'],$indexProperties[0]);
  1929.             $this->db->addIndex($indexType,$table,$indexName,$indexProperties);
  1930.                       
  1931.             break;
  1932.         } 
  1933.       }
  1934.     } 
  1935.     
  1936.     /*
  1937.      * creating link tables
  1938.      */
  1939.     $links = $doc->model->links;
  1940.     
  1941.     $this->log( "Model has " . ( is_object($links) ? count($links->children() ) : "no" ) . " links.", "propertyModel" );    
  1942.     
  1943.     if ( is_object($links) and count($links->children() ) )
