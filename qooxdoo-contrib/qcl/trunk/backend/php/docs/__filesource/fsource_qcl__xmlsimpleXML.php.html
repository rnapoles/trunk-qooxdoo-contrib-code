  682.   /**

  690.     {
  691.       $doc       =& $this->getDocument();

  709.     }
  710.   }
  711.   
  712.   /**
  713.    * extends one simple xml object tree with a second one
  714.    * currently, all nodes from the source tree are appended to
  715.    * the corresponding notes in the target tree. in a later version,
  716.    * the "extends" attribute will be used to selectively
  717.    * extend branches.
  718.    * @param object $target simple xml object (target)
  719.    * @param object $source simple xml object (source)
  720.    */
  721.   function _extend($target,$source)
  722.   {
  723.     /*
  724.      * iterate through source children and populate matching
  725.      * target children
  726.      */
  727.     $sourceChildren =& $source->children();
  728.     
  729.     for($i=0; $i<count($sourceChildren); $i++ )
  730.     {
  731.       $sourceChild =& $sourceChildren[$i];
  732.       
  733.       if ( ! is_object($sourceChild) )
  734.       {
  735.         $this->warn("Invalid source child:" . gettype($sourceChild));
  736.         continue;
  737.       }
  738.       
  739.       $tag = $sourceChild->getName();
  740.       $targetChild =& $target->$tag;      
  741.       //$this->debug("$tag -> " . gettype($targetChild) );
  743.       $doCopyNode  = true;      
  744.       
  745.       /*
  746.        * check if we have to copy or extend the node if we have
  747.        * two individual nodes
  748.        */
  749.       if ( is_object($sourceChild) and is_object($targetChild)  )
  750.       {
  751.         /*
  752.          * get attributes
  753.          */

  755.         $tgtChildAttrs = $targetChild->attributes();      
  756.   
  757.         if ( count($tgtChildAttrs) and count($srcChildAttrs) )
  758.         {
  759.           /*
  760.            * check if source 'extends' a 'name' attribute or if attributes are the same
  761.            * @todo Check if we can remove the "extend" stuff
  762.            */
  763.           if (
  764.             $tgtChildAttrs['extends'] == $srcChildAttrs['name'] or
  765.             $tgtChildAttrs == $srcChildAttrs
  766.           )
  767.           {
  768.             /*
  769.              * extend the node
  770.              */
  771.             //$this->debug("Extending tag $tag '{$tgtChildAttrs['extends']}'' with '{$tgtChildAttrs['name']}'.");
  772.             
  773.             $this->_extend(&$targetChild,&$sourceChild);
  774.             
  775.             /*
  776.              * add missing attributes
  777.              */
  778.             foreach($srcChildAttrs as $key => $value )
  779.             {
  780.               if ( ! isset($tgtChildAttrs[$key]) )
  781.               {
  782.                 $targetChild->addAttribute($key,$value);
  783.               }
  784.             }
  785.             $doCopyNode = false;          
  786.           }
  787.         }
  788.         elseif ( $tgtChildAttrs == $srcChildAttrs)
  789.         {
  790.           //$this->debug("Identical tag '$tag'' exists in target and source.");

  793.            */

  795.           $doCopyNode = false;
  796.         }
  797.         else
  798.         {
  799.           //$this->debug("Tag $tag exists in target and source but is not identical");
  800.         }
  801.       }
  802.       
  803.       /*
  804.        * copy source node(s) to target node
  805.        */

  807.       {
  808.         /*
  809.          * node to be copied over is a single node
  810.          */

  812.         {
  813.           /*
  814.            * source child attributes
  815.            */

  817.

  819.            * skip child node if target has a node that replaces 
  820.            * this node

  824.           {

  826.             continue;
  827.           }
  828.           
  829.           /*
  830.            * add new child to target node
  831.            */

  833.           $targetChild =& $target->addChild($tag,$cdata);
  834.           
  835.           //$this->debug("Creating single node $tag.");
  836.           
  837.           foreach( $srcChildAttrs as $key => $value )
  838.           {
  839.             $targetChild->addAttribute($key,$value);
  840.           }
  841.           
  842.           /*
  843.            * extend new node with possible source node children
  844.            */
  845.           $this->_extend(&$targetChild,&$sourceChild);
  846.         }
  847.  
  848.         elseif ( is_array( $sourceChild ) )
  849.         {
  850.           //$this->debug( count($sourceChild) . " nodes.");
  851.           /*
  852.            * we need to copy over an array of nodes
  853.            */
  854.           foreach ( $sourceChild as $s )
  855.           {
  856.             /*
  857.              * ignore non-object array members
  858.              */
  859.             if ( !is_object($s) )
  860.             {
  861.               $this->warn("No valid child node: " . gettype($s));
  862.               continue;
  863.             }
  864.  
  865.             /*
  866.              * source node attributes
  867.              */
  868.             $sAttr = $s->attributes();
  869.             
  870.             /*
  871.              * skip child node if target has a node that replaces 
  872.              * this node
  873.              */
  874.             $srcChildName = $sAttrs['name'];
  875.             //$this->debug("<$tag name='$name'>");
  876.             if ( $srcChildName and $this->getChildNodeByAttribute(&$target, "replace", $srcChildName ) )
  877.             {
  878.               //$this->debug("$srcChildName exists, not adding node...");
  879.               continue;
  880.             }
  881.             
  882.             /*
  883.              * add new child to target node 
  884.              */
  885.             $cdata = $this->getData(&$s);
  886.             //$this->debug("Creating sibling node $tag");
  887.             $targetChild =& $target->addChild($tag,$cdata);
  888.             
  889.             /*
  890.              * copy attributes to new node
  891.              */
  892.             foreach($s->attributes() as $key => $value )
  893.             {
  894.               $targetChild->addAttribute($key,$value);
  895.             }
  896.             
  897.             /*
  898.              * extend this new node with children of source
  899.              * node, if any
  900.              */
  901.             $this->_extend(&$targetChild,&$s);
  902.           }
  903.         }
  904.         else
  905.         {
  906.           /*
  907.            * ignore invalid source element
  908.            */ 
  909.           $this->warn("Invalid source element: " . gettype($sourceChild));
  910.         }
  911.       }
  912.     }
