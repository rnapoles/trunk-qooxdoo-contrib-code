  227.     /**

  234.     function StartElement($parser, $name, $attrs = array())
  235.     {   
  236.         //Check to see if tag is root-level
  237.         if (count($this->stack) == 0) 
  238.         {
  239.             //If so, set the document as the current tag
  240.             $this->document = new SimpleXmlElement($name, $attrs);
  241.             
  242.             //And start out the stack with the document tag
  243.             $this->stack = array('document');
  244.  
  245.             // taken from Ingo Schramm's simplexml44 Version 0.4.4 Ister.ORG 
  246.             // create a reference to doc to manipulate internal reference counter
  247.             // otherwise references won't work with nested arrays
  248.             // this is pretty a hack! (even works with 4.4.x)
  249.             // thanks to chat~kaptain524 at neverbox dot com at php.net
  250.             $r =& $this->document;        
  251.         }
  252.         //If it isn't root level, use the stack to find the parent
  253.         else
  254.         {
  255.             //Get the name which points to the current direct parent, relative to $this
  256.             $parent = $this->GetStackLocation();
  257.             
  258.             //Add the child
  259.             eval('
  260.               $parentNode =& $this->'.$parent.'; 
  261.               $child =& $parentNode->_AddChild($name, $attrs, '.count($this->stack).', $this->cleanTagNames);
  262.             ');
  263.             
  264.             //Added: Set id and parent id
  265.             $child->tagParentId = $parentNode->tagId;
  266.             
  267.             //Added: Save reference to node in numeric index
  268.             $this->nodeIndex[$child->tagId] =& $child;
  269.             
  270.             //Added: create index of tag names
  271.             $this->tagIndex[$name][] =& $child;
  272.  
  273.             //Added: create index of attributes. 
  274.             if ( count ( $this->indexedAttributes ) )
  275.             {              
  276.               foreach($this->indexedAttributes as $attrName)
  277.               {
  278.                 if ( isset($attrs[$attrName]) )
  279.                 {
  280.                   $value = $attrs[$attrName];
  281.                   $this->attributeIndex[$attrName][$value][] =& $child;
  282.                 }
  283.               }            
  284.             }
  285.             
  286.             //If the cleanTagName feature is on, replace colons and dashes with underscores
  287.             if($this->cleanTagNames)
  288.             {
  289.                 $name = str_replace(array(':', '-'), '_', $name);
  290.             }
  291.  
  292.             //Update the stack
  293.             eval('
  294.               $count = count($this->'.$parent.'->'.$name.'); 
  295.               $this->stack[] = $count > 1 ? $name.\'[\'.($count-1).\']\' : $name;
  296.             ');
  297.         }
  298.     }
