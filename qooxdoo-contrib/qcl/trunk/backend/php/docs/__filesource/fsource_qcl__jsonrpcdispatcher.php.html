  137. /*
  138.  * Error code, value -1: Script Error
  139.  *
  140.  * This error is raised when the service class aborts with an error,
  141.  * either syntax (parsing) errors or runtime exceptions.
  142.  *
  143.  * Although in PHP, it is possible to raise a custom error using the
  144.  * trigger_error() function, that technique is strongly discouraged when
  145.  * using JSON RPC; rather, return an appropriate application-specific error

  147.  * message.  An error generated with trigger_error() will always yield an
  148.  * error with origin=Server, code=ScriptError which makes it difficult for the
  149.  * client application to ascertain what went wrong.  This is particularly
  150.  * important when the application is to be localized, as it would never
  151.  * display a message directly from the server, but instead display its own

  219. function SendReply($reply, $scriptTransportId)
  220. {
  221.     /* If not using ScriptTransport... */
  222.     if ($scriptTransportId == ScriptTransport_NotInUse)
  223.     {
  224.         /* ... then just output the reply. */
  225.         print $reply;
  226.     }
  227.     else
  228.     {
  229.         /* Otherwise, we need to add a call to a qooxdoo-specific function */
  230.         $reply =
  231.             "qx.remote.ScriptTransport._requestFinished(" .
  232.             $scriptTransportId . ", " . $reply .
  233.             ");";
  234.         print $reply;
  235.     }
  236. }
  237.  
  238. /*
  239.  * Debug function. Define your own function if you want
  240.  * to do something else with the debug output
  241.  * @param string $str
  242.  */
  243. if ( ! function_exists("debug") )
  244. {
  245.   function debug($str)
  246.   {
  247.       static $fw = null;
  248.       if ($fw === null)
  249.       {
  250.           $fw = fopen("/tmp/phpinfo", "a");
  251.       }
  252.       fputs($fw, $str, strlen($str));
  253.       fflush($fw);
  254.   }
  255. }

  257. /*
  258.  * Start or join an existing session
  259.  */

  261.  
  262. /*
  263.  * Create a new instance of JSON and get the JSON-RPC request from
  264.  * the POST data.
  265.  */
  266. $json = new JSON();
  267. $error = new JsonRpcError($json);
  268.  
  269. /* Assume (default) we're not using ScriptTransport */
  270. $scriptTransportId = ScriptTransport_NotInUse;
  271.  
  272. if ($_SERVER["REQUEST_METHOD"] == "POST")
  273. {
  274.     /*
  275.      * For POST data, the only acceptable content type is application/json.
  276.      */
  277.     switch(substr($_SERVER["CONTENT_TYPE"], 
  278.         0, 
  279.         strcspn($_SERVER["CONTENT_TYPE"], ";")))
  280.     {
  281.     case "application/json":
  282.         /* We found literal POSTed json-rpc data (we hope) */
  283.         $input = file_get_contents('php://input');
  284.         $jsonInput = $json->decode($input);
  285.         break;
  286.     
  287.     default:
  288.         /*
  289.          * This request was not issued with JSON-RPC so echo the error rather
  290.          * than issuing a JsonRpcError response.
  291.          */
  292.         echo
  293.             "JSON-RPC request expected; " .
  294.             "unexpected data received";
  295.         exit;
  296.     }
  297. }
  298. else if ($_SERVER["REQUEST_METHOD"] == "GET" &&
  299.          isset($_GET["_ScriptTransport_id"]) &&
  300.          $_GET["_ScriptTransport_id"] != ScriptTransport_NotInUse &&
  301.          isset($_GET["_ScriptTransport_data"]))
  302. {
  303.     /* We have what looks like a valid ScriptTransport request */
  304.     $scriptTransportId = $_GET["_ScriptTransport_id"];
  305.     $error->SetScriptTransportId($scriptTransportId);
  306.     $input = $_GET["_ScriptTransport_data"];
  307.     $jsonInput = $json->decode(get_magic_quotes_gpc()
  308.                                ? stripslashes($input)
  309.                                : $input);
  310. }
  311. else
  312. {
  313.     /*
  314.      * This request was not issued with JSON-RPC so echo the error rather than
  315.      * issuing a JsonRpcError response.
  316.      */
  317.     echo "Services require JSON-RPC<br>";
  318.     exit;
  319. }
  320.  
  321. /* Ensure that this was a JSON-RPC service request */
  322. if (! isset($jsonInput) ||
  323.     ! isset($jsonInput->service) ||
  324.     ! isset($jsonInput->method) ||
  325.     ! isset($jsonInput->params))
  326. {
  327.     /*
  328.      * This request was not issued with JSON-RPC so echo the error rather than
  329.      * issuing a JsonRpcError response.
  330.      */
  331.     echo
  332.         "JSON-RPC request expected; " .
  333.         "service, method or params missing<br>";
  334.     exit;
  335. }
  336.  
  337. /*
  338.  * Ok, it looks like JSON-RPC, so we'll return an Error object if we encounter
  339.  * errors from here on out.
  340.  */
  341. $error->SetId($jsonInput->id);
  342.  
  343. /*
  344.  * Ensure the requested service name is kosher.  A service name should be:
  345.  *
  346.  *   - a dot-separated sequences of strings; no adjacent dots
  347.  *   - first character of each string is in [a-zA-Z] 
  348.  *   - other characters are in [_a-zA-Z0-9]
  349.  */
  350.  
  351. /* First check for legal characters */
  352. if (ereg("^[_.a-zA-Z0-9]+$", $jsonInput->service) === false)
  353. {
  354.     /* There's some illegal character in the service name */
  355.     $error->SetError(JsonRpcError_IllegalService,
  356.                      "Illegal character found in service name.");
  357.     $error->SendAndExit();
  358.     /* never gets here */
  359. }
  360.  
  361. /* Now ensure there are no double dots */
  362. if (strstr($jsonInput->service, "..") !== false)
  363. {
  364.     $error->SetError(JsonRpcError_IllegalService,
  365.                      "Illegal use of two consecutive dots in service name");
  366.     $error->SendAndExit();
  367. }
  368.  
  369. /* Explode the service name into its dot-separated parts */
  370. $serviceComponents = explode(".", $jsonInput->service);
  371.  
  372. /* Ensure that each component begins with a letter */
  373. for ($i = 0; $i < count($serviceComponents); $i++)
  374. {
  375.     if (ereg("^[a-zA-Z]", $serviceComponents[$i]) === false)
  376.     {
  377.         $error->SetError(JsonRpcError_IllegalService,
  378.                          "A service name component does not begin with a letter");
  379.         $error->SendAndExit();
  380.         /* never gets here */
  381.     }
  382. }
  383.  
  384. /*
  385.  * Now replace all dots with slashes so we can locate the service script.  We
  386.  * also retain the exploded components of the path, as the class name of the
  387.  * service is the last component of the path.
  388.  * == added CB: look for a folder "controllers" and append it if exists ==
  389.  */
  390. $servicePath = implode("/", $serviceComponents);
  391.  
  392. /* Try to load the requested service */
  393. if ( file_exists( servicePathPrefix . $servicePath . ".php") )
  394. {
  395.   require_once servicePathPrefix . $servicePath . ".php";
  396. }
  397. else
  398. {
  399.   array_splice($serviceComponents,count($serviceComponents)-1,0,array("controllers"));
  400.   $servicePath2 = implode("/", $serviceComponents);
  401.   if ( file_exists( servicePathPrefix . $servicePath2 . ".php") )
  402.   {
  403.     require_once servicePathPrefix . $servicePath2 . ".php";
  404.   }
  405.   else
  406.   {  
  407.       /* Couldn't find the requested service */
  408.       $error->SetError(JsonRpcError_ServiceNotFound,
  409.                        "Service `$servicePath` not found.");
  410.       $error->SendAndExit();
  411.       /* never gets here */
  412.   }
  413. }
  414.  
  415.  
  416. /* The service class is the last component of the service name */
  417. $className = JsonRpcClassPrefix . $serviceComponents[count($serviceComponents) - 1];
  418.  
  419. /* or the fully qualified service name */
  420. $longClassName = JsonRpcClassPrefix . implode("_", $serviceComponents );
  421.  
  422. /* Ensure that the class exists.  First try the short class name. */
  423. $classExists = class_exists($className);
  424. if (! $classExists)
  425. {
  426.     /* Short class name doesn't exist.  Try the long class name. */
  427.     $className = $longClassName;
  428.     $classExists = class_exists($className);
  429. }
  430.  
  431. if (! $classExists)
  432. {
  433.     $error->SetError(JsonRpcError_ClassNotFound,
  434.                      "Service class `" .
  435.                      $serviceComponents[count($serviceComponents) - 1] .
  436.                      "` not found.");
  437.     $error->SendAndExit();
  438.     /* never gets here */
  439. }
  440.  
  441. /* Instantiate the service */
  442. $service = new $className();
  443.  
  444. /*
  445.  * Do referer checking.  There is a global default which can be overridden by
  446.  * each service for a specified method.
  447.  */
  448.  
  449. /* Assign the default accessibility */
  450. $accessibility = defaultAccessibility;
  451.  
  452. /*
  453.  * See if there is a "GetAccessibility" method in the class.  If there is, it
  454.  * should take two parameters: the method name and the default accessibility,
  455.  * and return one of the Accessibililty values.
  456.  */
  457. if (method_exists($service, "GetAccessibility"))
  458. {
  459.     /* Yup, there is.  Get the accessibility for the requested method */
  460.     $accessibility = $service->GetAccessibility($jsonInput->method,
  461.                                                 $accessibility);
  462. }
  463.  
  464. /* Do the accessibility test. */
  465. switch($accessibility)
  466. {
  467.   case Accessibility_Public:
  468.     /* Nothing to do.  The method is accessible. */
  469.     break;
  470.     
  471.   case Accessibility_Domain:
  472.     /* Determine the protocol used for the request */
  473.     if (isset($_SERVER["SSL_PROTOCOL"]))
  474.     {
  475.         $requestUriDomain = "https://";
  476.     }
  477.     else
  478.     {
  479.         $requestUriDomain = "http://";
  480.     }
  481.  
  482.     // Add the server name
  483.     $requestUriDomain .= $_SERVER["SERVER_NAME"];
  484.  
  485.     // The port number optionally follows.  We don't know if they manually
