  504.        * - the lock is older than 600 seconds (no process is supposed to be longer than 10 minutes)
  505.        */
  506.       if ( ! $lock or ( 
  507.         is_array($lock) and ( 
  508.           $lock['session_id'] == session_id() or ( $time - $lock['timestamp'] > 600 ) 
  509.         ) ) )
  510.       {      
  511.         $hasLock = true;
  512.         /*
  513.          * create new lock with session id and timestamp
  514.          */
  515.         $lock = array(
  516.           'session_id' => session_id(),
  517.           'timestamp'  => $time
  518.         );
  519.         $this->store($lockId, $lock );
  520.         break;
  521.       }
  522.       /*
  523.        * sleep for 100 ms to give the cpu a break
  524.        */
  525.       usleep(100);
  526.     }
  527.     if ( ! $hasLock )
  528.     {
  529.       $this->raiseError( get_class($this) . ": cannot get lock for '$lockId' since 5 seconds . Aborting.");
  530.     }
  531.     return;
  532.   }

  539.    */
  540.   function removeLock ( $lockId )

  550.     $lockId = md5($lockId) . ".lock";    
  551.     $this->remove($lockId);
  552.   }

  554.   /**

  556.    * returns true on first check and throws an error
